# API ROUTES

## Accounts:
  GET: /accounts
      /accounts/:id

  PUT: /accounts/:id

  DELETE: /accounts/:id

  POST: /accounts

## Transactions:
  GET: /transactions
       /transactions/:id

PUT: /transactions/:id

DELETE: /transactions/:id

POST: /transactions

## Cards:
  GET: /cards
       /cards/:id
       /cards/invoices/:id

PUT: /cards/:id

DELETE: /cards/:id

POST: /cards

## Invoices:
  GET: /invoices
       /invoices/:id
       /invoices/transactions/:id

PUT: /invoices/:id

DELETE: /invoices/:id

POST: /invoices
